Пример использования:

Tested_t Example(_T("\\b(sub)([^ ]*)"));
Example.IsSearch(_T("this subject has a submarine as a subsequence"));
auto Coincided = Example.GetCoincided();
ASSERT_EQ(2, Coincided[_T("subject")].size());
EXPECT_TRUE(Coincided[_T("subject")][0] == _T("sub"));
EXPECT_TRUE(Coincided[_T("subject")][1] == _T("ject"));
ASSERT_EQ(2, Coincided[_T("submarine")].size());
EXPECT_TRUE(Coincided[_T("submarine")][0] == _T("sub"));
EXPECT_TRUE(Coincided[_T("submarine")][1] == _T("marine"));
ASSERT_EQ(2, Coincided[_T("subsequence")].size());
EXPECT_TRUE(Coincided[_T("subsequence")][0] == _T("sub"));
EXPECT_TRUE(Coincided[_T("subsequence")][1] == _T("sequence"));
