Класс входит в Проект Testing
Базовый класс для построения прокси-классов для проверки вызовов функций подставных классов.

Предупреждения
Класс не следует использовать для проверки работоспособности в многопоточной среде.

Пример использования:

// Создаем подставной класс с вызовами подставных функций прокси-класса:
namespace mock
{
//
// Создаем прокси-класс с нужными подставными функциями:
class ProjectProxy : public ::alicorn::testing::Proxy<ProjectProxy>
{
public:
MOCK_METHOD1(Constructor, size_t(int));
MOCK_CONST_METHOD1(GetName, CString(size_t));
};
//
class Project
{
public:
const size_t m_Id;
public:
CString GetName(void) const
{
return ProjectProxy::GetInstance()->GetName(m_Id);
}
public:
Project(int _Index) :
m_Id(ProjectProxy::GetInstance()->Constructor(_Index))
{
}
};
//
} // namespace mock
//
// Тестируемая функция:
class Example
{
public:
CString GetProjectNameFromIndex(int _Index)
{
// В тестовом проекте здесь должен использоваться класс mock::Project.
Project Project(_Index);
return Project.GetName();
}
};
//
// Использование прокси-класса в тесте:
TEST_F(Test_test, Test_GetName)
{
const int Index = 1;
const size_t ProjectId = 1509032221;
const CString ExpectedName = _T("Name");
typedef mock::ProjectProxy ProjectProxy_t;
ProjectProxy_t ProjectProxy;
ProjectProxy_t::GetInstance() = &ProjectProxy;
using namespace ::testing;
InSequence Dummy;
EXPECT_CALL(ProjectProxy, Constructor(Index))
.Times(1)
.WillOnce(Return(ProjectId));
EXPECT_CALL(ProjectProxy, GetName(ProjectId))
.Times(1)
.WillOnce(Return(ExpectedName));
Example Example;
const CString Result = Example.GetProjectNameFromIndex(Index);
EXPECT_STREQ(ExpectedName, Result);
}
