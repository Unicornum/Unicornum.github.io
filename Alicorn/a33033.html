Примеры использования

Работа со строками

Исходные данные:

using String_t = ::alicorn::extension::std::String;

Конструирование объектов:

const auto ExampleUtf8_1 = u8T("Текст Text");
const auto ExampleUtf8_2(u8T("Текст Text"));
const auto ExampleUtf8_3(u8T(R"---(Текст Text)---"));
const auto ExampleUsc2_1 = uT("Текст Text");
const auto ExampleUsc2_2(uT("Текст Text"));
const auto ExampleUsc2_3(uT(R"---(Текст Text)---"));
const auto ExampleUtf8 = u8T("Текст Text");
const auto ExampleUsc2 = uT("Текст Text");
const auto CopyUtf8_1 = ExampleUtf8;
const auto CopyUtf8_2(ExampleUtf8);
const auto CopyUsc2_1 = ExampleUsc2;
const auto CopyUsc2_2(ExampleUsc2);

Копирование объектов:

const auto SourceUtf8 = u8T("Текст Text");
const auto SourceUsc2 = uT("Текст Text");
String_t Example;
Example = SourceUtf8;
EXPECT_STREQ(u8"Текст Text", Example.ToRawString<char>());
Example = SourceUsc2;
EXPECT_STREQ(L"Текст Text", Example.ToRawString<wchar_t>());

Количество символов в строке:

const auto LenghtUtf8 = u8T("Текст Text").GetLenght();
EXPECT_EQ(15, LenghtUtf8); // Не количество букв, а количество символов
const auto LenghtUsc2 = uT("Текст Text").GetLenght();
EXPECT_EQ(10, LenghtUsc2);

Удаление пробельных символов:

const auto SourceUsc2 = uT(" \t\r\nТекст Text \t\r\n");
EXPECT_STREQ(L"Текст Text", SourceUsc2.Trim().ToRawString<wchar_t>());

Выделение подстроки:

const auto SourceUtf8 = u8T("123 Текст Text");

Сложение строк:

const auto SourceUtf8_1 = u8T("ГДЕ");
const auto SourceUtf8_2 = u8T("ЁЖЗ");
const auto SourceUsc2_1 = uT("ГДЕ");
const auto SourceUsc2_2 = uT("ЁЖЗ");
// Формат результата определяется форматом первой строки
const auto ResultUtf8 =
u8T("АБВ ") +
SourceUsc2_1 + // Предварительное преобразование в Utf8
u8T(" ") +
SourceUtf8_2;
EXPECT_STREQ(u8"АБВ ГДЕ ЁЖЗ", ResultUtf8.ToRawString<char>());
// Формат результата определяется форматом первой строки
const auto ResultUsc2 =
uT("АБВ ") +
SourceUtf8_1 + // Предварительное преобразование в Usc2
uT(" ") +
SourceUsc2_2;
EXPECT_STREQ(L"АБВ ГДЕ ЁЖЗ", ResultUsc2.ToRawString<wchar_t>());
const auto SourceUtf8_1 = u8T("ГДЕ");
const auto SourceUtf8_2 = u8T("ЁЖЗ");
const auto SourceUsc2_1 = uT("ГДЕ");
const auto SourceUsc2_2 = uT("ЁЖЗ");
String_t ResultUtf8 = u8T("");
ResultUtf8 += u8T("АБВ ");
ResultUtf8 += SourceUsc2_1; // Предварительное преобразование в Utf8
ResultUtf8 += u8T(" ");
ResultUtf8 += SourceUtf8_2;
// Формат результата определяется форматом исходной строки
EXPECT_STREQ(u8"АБВ ГДЕ ЁЖЗ", ResultUtf8.ToRawString<char>());
String_t ResultUsc2; // Конструктор по умолчанию создает строку Usc2
ResultUsc2 += uT("АБВ ");
ResultUsc2 += SourceUtf8_1; // Предварительное преобразование в Usc2
ResultUsc2 += uT(" ");
ResultUsc2 += SourceUsc2_2;
// Формат результата определяется форматом исходной строки
EXPECT_STREQ(L"АБВ ГДЕ ЁЖЗ", ResultUsc2.ToRawString<wchar_t>());

Замена подстроки:

Преобразование Unicode строк без изменения формата

using namespace ::alicorn::extension::std;
const auto Source = uT("Текст Text"); // можно u8T(), u16T() или u32T() -
// это не имеет значения
const auto Utf8String = string_cast<::std::string>(Source);
const auto Ucs2String = string_cast<::std::wstring>(Source);
const auto Utf16String = string_cast<::std::u16string>(Source);
const auto Utf32String = string_cast<::std::u32string>(Source);
// Во всех случаях ниже объект строки результата будет содержать внутри себя
// копию исходной строки в формате исходной строки (для изменения формата
// следует использовать вариант с локалью (см. ниже)).
const auto Utf8Example = string_cast<String>(Utf8String);
const auto Ucs2Example = string_cast<String>(Ucs2String);
const auto Utf16Example = string_cast<String>(Utf16String);
const auto Utf32Example = string_cast<String>(Utf32String);

Преобразование строк с локалью

using namespace ::alicorn::extension::std;
const auto Source = uT("Текст Text"); // можно u8T(), u16T() или u32T() -
// это не имеет значения
const auto Cp1251Example = string_cast<::std::string, Locale::Russian::CP1251>(Source);
const auto Cp866Example = string_cast<::std::string, Locale::Russian::CP866>(Source);
// Установка глобальной локали по умолчанию:
// Теперь во всем коде текущего модуля будет использоваться
// кодовая страница ru_RU.CP1251
Locale::Global::SetDefault<Locale::Russian::CP1251>();
const auto Cp1251Example = string_cast<String, Locale::Default>(Source);
{
// Установка локали по умолчанию для текущей области видимости:
// Кодовая страница ru_RU.CP866 будет использоваться до закрывающей
// фигурной скобки
Locale::Scope::SetDefault<Locale::Russian::CP866> Dummy;
const auto Cp866Example = string_cast<String, Locale::Default>(Source);
}
// Использование локали по умолчанию для работы с потоками.
::std::wstringstream Stream;
Stream << Locale::Default() << 123.456; // В поток запишется "123,456"

Функция входит в Std
Функция преобразования строк из одной кодировки в другую.

  • Вспомогательный шаблон для преобразования String в/из различных форматов UTF (поддерживаются UTF8, UCS2, UTF16, UTF32).
  • Функция предназначена для поддержки работы класса alicorn::extension::std::String и позволяет производить следующие виды преобразований (не поддерживаемые не будут компилироваться):
  • String <-> std::u32string.
  • String <-> std::u16string.
  • String <-> std::wstring.
  • String <-> std::string(utf8).
  • String <-> boost::filesystem::path (требуется включение файла <alicorn\extension\boost\string-cast.hpp>).
  • String <-> CString (требуется включение файла <alicorn\extension\mfc\string-cast.hpp>).
  • String <-> wxString (требуется включение файла <alicorn\extension\wx.hpp>).

TTargetКласс строки, которую нужно получить.

Функция входит в Std
Функция преобразования строк из одной кодировки в другую.

  • Функция представляет собой шаблон, предоставляющий интерфейс для преобразования строк класса String в/из объектов класса std::string в указанной кодировке.
  • Функция предназначена для поддержки работы класса alicorn::extension::std::String и позволяет производить следующие виды преобразований (для несуществующих будет брошено исключение):
  • String <-> std::string(1251).
  • String <-> std::string(866).
  • String <-> std::string(utf8).
Параметры шаблона
TTargetКласс строки, которую нужно получить.
TLocaleКласс локали используемой кодировки (при преобразовании std::string в String это кодировка исходной строки, при обратном - кодировка результата).
Для задания кодировки использовать дочерние классы класса Locale.
