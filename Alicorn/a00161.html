Пример использования:

const ::std::vector<String_t> Expected =
{
uT("ject"),
uT("sequence"),
uT("ject"),
uT("marine"),
};
Tested_t Example(uT("\\b(sub)([^ ]*)"));
Example.IsSearch(uT("this subject has a subsequence as a subject submarine"));
auto Coincided = Example.GetCoincided();
ASSERT_EQ(Expected.size(), Coincided.size());
size_t i = 0;
for (const auto & Value : Coincided)
{
EXPECT_EQ(uT("sub") + Expected[i], Value.first);
ASSERT_EQ(2, Value.second.size());
EXPECT_EQ(uT("sub"), Value.second[0]);
EXPECT_EQ(Expected[i], Value.second[1]);
i++;
}
