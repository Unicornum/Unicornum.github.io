91{
92 auto & CreationFunctions = FactoryRef_t::GetInstance().m_CreationFunctions;
93 const auto itFunction = CreationFunctions.find(_ClassId);

115template<class TInterface, class TId, class ... TArgs>
116auto Factory<TInterface, TId, TArgs ...>::GetCreateFunction(
117 const TId & _ClassId) const -> Creations_t
118{
119 auto itClassCreationFunction = m_CreationFunctions.find(_ClassId);
120 if (itClassCreationFunction == m_CreationFunctions.end())
121 {
122 throw STD_EXCEPTION <<
123 "Class '" << _ClassId << "' not registered [" <<
124 "interface: " << TypeName<TInterface>::Get() << ", " <<
125 "id: " << TypeName<TId>::Get() << "].";
126 }
127
128 return itClassCreationFunction->second;
129}
