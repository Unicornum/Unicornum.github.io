Пример использования:

Tested_t Example(_T("([1234]+)-([1234]+)-([1234]+)"));
Example.IsMatch(_T("123-234-341"));
const auto Groups = Example.GetGroups();
ASSERT_EQ(3, Groups.size());
EXPECT_TRUE(Groups[0] == _T("123"));
EXPECT_TRUE(Groups[1] == _T("234"));
EXPECT_TRUE(Groups[2] == _T("341"));
