Использование xml документов
#include <alicorn\document.hpp>
using namespace ::alicorn::source::document;
// Исходные определения
using String_t = ::alicorn::extension::std::String;
using namespace ::alicorn::source;
using XmlDefault_t = details::Universal<details::Default, Universal<RapidXml>>;
// Это не скомпилируется, с xml документами можно работать ТОЛЬКО в формате utf8
//const XmlDefault_t Example;
// Создание нового документа
Xml_t Example;
EXPECT_FALSE(Example.IsContainsChildNodes());
auto Root = Example[u8T("root")];
EXPECT_TRUE(Example.IsContainsChildNodes());
EXPECT_FALSE(Root.IsContainsChildNodes());
auto First = Root[u8T("first")];
First.Value = u8T("Value1");
First.Attributes[u8T("param")] = u8T("12345");
First.Attributes[u8T("is_true")] = true;
auto Root2 = First[Parent];
auto Second = Root[u8T("second")];
Second.Value = 67890;
Second.Attributes[u8T("param")] = 54321;
Second.Attributes[u8T("is_false")] = false;
EXPECT_TRUE(Root.IsContainsChildNodes());
Example.Save(m_PathToResultFile);
// Результат:
<?xml version="1.0" encoding="utf-8"?>
<root>
<first param="12345" is_true="true">Value1</first>
<second param="54321" is_false="false">67890</second>
</root>
// Использование узла после удаления объекта документа
class Support
{
public:
static auto GetNodeFromName(const String_t & _Name)
{
const Xml_t Example{ THIS_DIRECTORY / L"Example.xml" };
for (const auto & ChildNode : Example[u8T("root")])
{
if (ChildNode.GetId() == _Name) return ChildNode;
}
throw ::std::exception("Node not found.");
}
};
auto First = Support::GetNodeFromName(u8T("first"));
EXPECT_EQ(u8T("first"), First.GetId());
EXPECT_EQ(u8T("Value1"), First.Value.To<String_t>());
EXPECT_EQ(u8T("12345"), First.Attributes[u8T("param")].To<String_t>());
EXPECT_EQ(u8T("true"), First.Attributes[u8T("is_true")].To<String_t>());
EXPECT_EQ(true, First.Attributes[u8T("is_true")].To<bool>());
// Чтение документа из файла
const Xml_t Example{ THIS_DIRECTORY / L"Example.xml" };
EXPECT_THROW(Example[u8T("unknown")], ::std::exception);
auto Root = Example[u8T("root")];
EXPECT_EQ(u8T("root"), Root.GetId());
EXPECT_EQ(u8T(""), Root.Value.To<String_t>());
EXPECT_THROW(Root.Attributes[u8T("unknown")].To<String_t>(), ::std::exception);
class NodeData
{
public:
String_t m_Value;
String_t m_Param;
};
::std::map<String_t, NodeData> Expected =
{
{ u8T("first"), { u8T("Value1"), u8T("12345") } },
{ u8T("second"),{ u8T("67890"), u8T("54321") } },
};
auto NodeCount = 0;
for (const auto & ChildNode : Root)
{
const auto Name = ChildNode.GetId();
EXPECT_EQ(Expected[Name].m_Value, ChildNode.Value.To<String_t>());
EXPECT_EQ(Expected[Name].m_Param, ChildNode.Attributes[u8T("param")].To<String_t>());
NodeCount++;
}
EXPECT_EQ(2, NodeCount);
// Поиск узла по имени...
const Xml_t Example{ THIS_DIRECTORY / L"Example.xml" };
auto Root = Example[u8T("root")];
// ... перебором дочерних узлов
Xml_t::Node_t First;
EXPECT_THROW(First.GetId(), ::std::exception);
for (const auto & ChildNode : Root)
{
if (ChildNode.GetId() == u8T("first")) First = ChildNode;
}
EXPECT_EQ(u8T("first"), First.GetId());
EXPECT_EQ(u8T("Value1"), First.Value.To<String_t>());
EXPECT_EQ(u8T("12345"), First.Attributes[u8T("param")].To<String_t>());
EXPECT_EQ(u8T("true"), First.Attributes[u8T("is_true")].To<String_t>());
EXPECT_EQ(true, First.Attributes[u8T("is_true")].To<bool>());
// ... при помощи стандартного алгоритма
auto itSecond = ::std::find_if(Root.begin(), Root.end(),
[](const auto & _Node)
{
return (_Node.GetId() == u8T("second"));
});
ASSERT_TRUE(itSecond != Root.end());
auto Second = *itSecond;
EXPECT_EQ(u8T("second"), Second.GetId());
EXPECT_EQ(u8T("67890"), Second.Value.To<String_t>());
EXPECT_EQ(54321, Second.Attributes[u8T("param")].To<int>());
EXPECT_EQ(u8T("false"), Second.Attributes[u8T("is_false")].To<String_t>());
EXPECT_EQ(false, Second.Attributes[u8T("is_false")].To<bool>());

Использование вложенных документов

Если xml документ содержит узел link (или a), в качестве атрибута href которого указан путь к другому документу xml (относительно текущего документа), то содержимое этого документа будет подгружено как дочерний узел узла link.

Например, документы, ссылающиеся друг на друга следующим образом:

<?xml version="1.0" encoding="utf-8"?>
<root>
<link href="Child.xml"/>
<first/>
</root>
<?xml version="1.0" encoding="utf-8"?>
<root_child>
<second/>
<link href="Child\Child.xml"/>
</root_child>
<?xml version="1.0" encoding="utf-8"?>
<root_child_child>
<third/>
</root_child_child>

равносильны одному документу со следующей структурой:

<?xml version="1.0" encoding="utf-8"?>
<root>
<link href="Child.xml">
<root_child>
<second/>
<link href="Child\Child.xml">
<root_child_child>
<third/>
</root_child_child>
</link>
</root_child>
</link>
<first/>
</root>
