Подробное описание

Раздел классов доступа к данным изображений.

Примеры использования

Исходное изображение

Изображение увеличено, каждый квадрат - отдельный пиксель.

Простое чтение данных изображения

using namespace ::alicorn::source;
// Простое чтение данных изображения:
// - Длина строки изображения в байтах будет (ширина * 3).
// - Строки изображения будут идти сверху вниз.
// - Байты пикселя будут идти в порядке BGR.
const image::Universal_t<image::pixel::BGR> Example
{
::boost::filesystem::load_binary_file(PathToSourceFile)
};
const auto Result = Example.GetData();
EXPECT_TRUE(Result.Bpp == 3);
EXPECT_TRUE(Result.Width == 3);
EXPECT_TRUE(Result.Height == 2);
EXPECT_TRUE(Result.Stride == 9);
const RawData_t ExpectedData =
{
0x00, 0x00, 0xFF, 0x00, 0xFF, 0x00, 0xFF, 0x00, 0x00,
0xFF, 0x00, 0xFF, 0xFF, 0xFF, 0x00, 0x00, 0xFF, 0xFF,
};
EXPECT_TRUE(Result.Buffer == ExpectedData);

Чтение с применением модификаторов

using namespace ::alicorn::source;
// Использование нескольких модификаторов обновременно:
// - Байты пикселя будут идти в порядке RGBA.
const image::Universal_t<image::pixel::RGBA> Example
{
image::Stride<5> // Ширина изображения будет кратна 5 байтам
{
image::InvertLine // Строки изображения будут идти снизу вверх
{
::boost::filesystem::load_binary_file(PathToSourceFile)
}
}
};
const auto Result = Example.GetData();
EXPECT_TRUE(Result.Bpp == 4);
EXPECT_TRUE(Result.Width == 3);
EXPECT_TRUE(Result.Height == 2);
EXPECT_TRUE(Result.Stride == 15);
const RawData_t ExpectedData =
{
0xFF, 0x00, 0xFF, 0xFF, 0x00, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0x00, 0xFF,
0x00, 0x00, 0x00,
0xFF, 0x00, 0x00, 0xFF, 0x00, 0xFF, 0x00, 0xFF, 0x00, 0x00, 0xFF, 0xFF,
0x00, 0x00, 0x00,
};
EXPECT_TRUE(Result.Buffer == ExpectedData);
