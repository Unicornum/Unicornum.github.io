Класс входит в Patterns
Фабрика объектов (создание объектов по идентификатору (строке) класса).

Доработанный вариант, взятый отсюда .

Параметры шаблона
TInterfaceКласс интерфейса, в указателях на который фабрика будет создавать объекты.
TIdКласс идентификатора класса, при помощи которых будут создаваться объекты (требеутся оператор 'ostream <<' для добавления идентификатора в сообщение об ошибке).

Пример использования:

class IBase
{
public:
virtual void func() const = 0;
virtual ~IBase() {}
};
class Base : public IBase
{
public:
void func() const override { std::cout << "Base::func\n"; }
};
class Der1 : public Base
{
public:
void func() const override { std::cout << "Der1::func()" << std::endl; }
};
FACTORY_REGISTER_STRING_NAME(Der1, IBase);
class Der2 : public Base
{
public:
void func() const override { std::cout << "Der2::func()" << std::endl; }
};
FACTORY_REGISTER_STRING_NAME(Der2, IBase);
int main(void)
{
using Factory_t = ::alicorn::patterns::factory::Factory<IBase>;
auto pObject = Factory_t::Create("Der1");
if(pObject) pObject->func(); // 'Der1::func()'
pObject = Factory_t::Create("Der2");
if(pObject) pObject->func(); // 'Der2::func()'
return 0;
}
