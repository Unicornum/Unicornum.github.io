Класс входит в проект Std
Простой класс-обертка для указателя.

  • Простая обертка над указателем, который инициализируется в конструкторе и гарантированно уничтожается в деструкторе.
  • Понадобился из-за того, что при использовании std::unique_ptr<> нужно указывать в заголовочном файле "#include <memory>";
  • Должен использоваться исключительно для реализации паттерна pImpl.
Параметры шаблона
TКласс объекта, который будет содержаться внутри Ptr.
Заметки
Если требуется поддержка копирования, должен содержать функцию, возвращающую указатель на созданную оператором new копию объекта
class RealClass
{
public:
RealClass * Clone(void) const { return new RealClass(*this); }
...

Using.cpp

#include "stdafx.h"
#include "Example.hpp"
// Не требуется ни объявление, ни реализация класса Implementation
void Using(void)
{
const Tested Example(1702071213);
const auto Value = Example.GetValue();
if (Value != 1702071213) throw ::std::exception("Unexpected value.");
const Tested Added = Example.Add(1111111111);
const auto AddedValue = Added.GetValue();
if (AddedValue != 2813182324) throw ::std::exception("Unexpected value.");
}

Implementation.hpp

#pragma once
#include "Example.hpp"
class ImplementationProxy :
public ::alicorn::testing::Proxy<ImplementationProxy>
{
public:
MOCK_METHOD1(Constructor, int(int));
MOCK_METHOD1(Destructor, void(int));
};
// cppcheck-suppress noConstructor
class Tested::Implementation :
public IImplementation
{
public:
int GetValue(void) const override
{
return m_Value;
}
Impl_t Add(int _Source) const override
{
return Impl_t::Make<Implementation>(m_Value + _Source);
}
IImplementation * Clone(void) const override
{
return new Implementation(m_Value);
}
private:
const int m_Value = 0;
const int m_Id;
public:
explicit Implementation(int _Value) :
m_Value(_Value),
m_Id(ImplementationProxy::GetInstance()->Constructor(_Value))
{
}
~Implementation(void)
{
ImplementationProxy::GetInstance()->Destructor(m_Id);
}
};
