Набор вспомогательных библиотек

Запуск программы без параметров и без ожидания ее завершения

::alicorn::system::platform::Process{ "calc.exe" }.Run();

Запуск программы с параметрами и ожиданием ее завершения

::alicorn::system::platform::Process{ "calc.exe" }.Wait(1000).Run({ uT("--help") });

Запуск программы с получением ее консольного вывода

::std::string ConsoleOutput;
auto Receiver = [&](const ::std::string & _Line)
{
ConsoleOutput += _Line + "\r\n";
};
::alicorn::system::platform::Process{ "calc.exe" }.StdOut(Receiver).Run();
