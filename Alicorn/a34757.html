Пример использования:

Tested_t Example(uT("([1234]+)-([1234]+)-([1234]+)"));
Example.IsMatch(uT("123-234-341"));
const auto Groups = Example.GetGroups();
ASSERT_EQ(3, Groups.size());
EXPECT_EQ(uT("123"), Groups[0]);
EXPECT_EQ(uT("234"), Groups[1]);
EXPECT_EQ(uT("341"), Groups[2]);
