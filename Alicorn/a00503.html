Подробное описание

Раздел классов доступа к настройкам программы.

Примеры использования:

Образец файла настроек:

1 <?xml version="1.0" encoding="utf-8"?>
2 <root>
3  <!-- Раздел настроек -->
4  <settings>
5  <!-- Общие настройки для всех остальных разделов -->
6  <Common parent="">
7  <IsTrue>true</IsTrue>
8  <IsFalse>false</IsFalse>
9  <English>English</English>
10  <Russian>Русский</Russian>
11  <String>Empty</String>
12  <Number>123</Number>
13  <Reference1>Number: {Number}</Reference1>
14  <Reference2>Param: {Unknown}</Reference2>
15  <Reference3>Number: {Number} {Number} {String}</Reference3>
16  <Reference4>Number: {Number} {Unknown} {String}</Reference4>
17  <Param1>Unknown</Param1>
18  <First parent="">
19  <Param1>First</Param1>
20  </First>
21  <Second parent="">
22  <Param1>First</Param1>
23  <Param2>Number: {Number} {Param1}</Param2>
24  </Second>
25  </Common>
26  <!-- Раздел, ссылающийся на раздел Common (существующие в разделе -->
27  <!-- параметры будут загружены как есть, после чего будут добавлены -->
28  <!-- отсутствующие параметры из родительского раздела). -->
29  <Current parent="Common">
30  <Number>234</Number>
31  </Current>
32  <!-- Раздел, ссылающийся на раздел Current -->
33  <Adding parent="Current">
34  <String>Using string</String>
35  <Int>12345</Int>
36  </Adding>
37  </settings>
38 </root>

Работа с параметрами:

// Создание объекта раздела Adding
auto AddingSection = ::alicorn::settings::RootSection()[_T("Adding")];
// Создание объекта подраздела Adding\Second
auto AddinSecondSection = AddingSection[_T("Second")];
// Создание нового раздела
auto AddinNewSection = AddingSection[_T("New")];
//
// Установка значений по умолчанию
AddingSection.SetDefault(_T("Bool"), true); // не существует, будет добавлен
AddingSection.SetDefault(_T("Bool2"), _T("unknown")); // не существует, будет добавлен
AddingSection.SetDefault(_T("Number"), 999); // уже существует, не изменится
//
AddinNewSection.SetDefault(_T("One"), 1);
AddinNewSection.SetDefault(_T("Two"), _T("2"));
AddinNewSection.SetDefault(_T("Three"), _T("333"));
//
// Установка существующих параметров (будут заменены)
AddingSection.Set(_T("String"), _T("New string"));
AddingSection.Set(_T("Bool2"), true);
AddinNewSection.Set(_T("Three"), 3);
//
// Установка не существующих параметров (исключение)
EXPECT_THROW(AddingSection.Set(_T("Unknown"), _T("")), std::exception);
//
// Чтение особых параметров
EXPECT_EQ(::alicorn::system::CurrentModule::GetLocation(),
AddingSection.Get<Path_t>(_T("ModuleLocation")));
//
// Чтение существующих параметров
EXPECT_EQ(true, AddingSection.Get<bool>(_T("IsTrue")));
EXPECT_EQ(false, AddingSection.Get<bool>(_T("IsFalse")));
EXPECT_EQ(234, AddingSection.Get<int>(_T("Number")));
EXPECT_EQ(12345, AddingSection.Get<int>(_T("Int")));
EXPECT_EQ(true, AddingSection.Get<bool>(_T("Bool"))); // добавленный
EXPECT_EQ(true, AddingSection.Get<bool>(_T("Bool2"))); // добавленный и измененный
EXPECT_STREQ(_T("New string"),
AddingSection.Get<String_t>(_T("String")).ToRawString()); // измененный
EXPECT_EQ(1, AddinNewSection.Get<int>(_T("One")));
//
// Чтение параметров, содержащих ссылки на другие параметры
EXPECT_STREQ(_T("Number: 234 234 New string"),
AddingSection.Get<String_t>(_T("Reference3")).ToRawString());
EXPECT_STREQ(_T("Number: 234 {Unknown} New string"),
AddingSection.Get<String_t>(_T("Reference4")).ToRawString());
//
// Чтение не существующих параметров (исключение)
EXPECT_THROW(AddingSection.Get<String_t>(_T("Unknown")), std::exception);
EXPECT_THROW(AddinNewSection.Get<String_t>(_T("Five")), std::exception);
//
// Чтение с указанием невалидного типа (исключение)
EXPECT_THROW(AddingSection.Get<bool>(_T("String")), std::exception);
EXPECT_THROW(AddingSection.Get<int>(_T("String")), std::exception);
