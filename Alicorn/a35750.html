Использование текстовых документов
#include <alicorn\document.hpp>
using namespace ::alicorn::document;
// Исходные определения
using String_t = ::alicorn::extension::std::String;
// Создание нового документа, заполнение его строками и сохранение в файл
Text_t Example;
Example += uT("Text1");
Example += uT("Text2");
Example += uT("Text3");
Example.Save(m_PathToResultFile);
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
// Чтение документа из файла, добавление строк и сохранение в исходный файл
Text_t Example(m_PathToResultFile);
Example += uT("Text4");
Example += uT("Text5");
Example.Save();
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
// Text4
// Text5
// Доступ к строкам текста по индексу
Text_t Example;
Example += uT("Text1");
Example += uT("Text2");
Example += uT("Text3");
EXPECT_EQ(3, Utilities::GetLinesCount(Example));
EXPECT_EQ(uT("Text1"), Example[0].Value.To<String_t>());
EXPECT_EQ(uT("Text2"), Example[1].Value.To<String_t>());
EXPECT_EQ(uT("Text3"), Example[2].Value.To<String_t>());
EXPECT_STDEXCEPTION(Example[uT("")],
".+Action does not make sense.+");
EXPECT_STDEXCEPTION(Example[1703010913],
".+Action does not make sense.+");
// Проход по всем строкам документа и замена '{INDEX}' в каждой строке на очередной индекс
Text_t Example;
Example += uT("Text{INDEX}");
Example += uT("Text{INDEX}");
Example += uT("Text{INDEX}");
int Index = 0;
for (auto Line : Example)
{
Line.Value = Line.Value.To<String_t>().Replace(uT("{INDEX}"), ++Index);
}
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
// Последовательный поиск строки и замена найденной строки на новую
Text_t Example;
Example += uT("Text???");
Example += uT("Text???");
Example += uT("Text???");
auto itFindedLine = Example.begin();
for (const auto & Text : { uT("Text1"), uT("Text2"), uT("Text3") })
{
itFindedLine = ::std::find_if(itFindedLine, Example.end(),
[](const auto & _Line)
{
return (_Line.Value.To<String_t>() == uT("Text???"));
});
ASSERT_TRUE(itFindedLine != Example.end());
(*itFindedLine).Value = Text;
}
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
// Вставка набора строк после каждой строки 'Text'
Text_t Example;
Example += uT("Text");
Example += uT("Text4");
Example += uT("Text");
Example = Utilities::Insert(Example, { uT("Text2"), uT("Text3") },
[](const auto & _Line)
{
return (_Line.Value.To<String_t>() == uT("Text"));
});
// В результате будет текстовый документ, содержащий
// Text
// Text2
// Text3
// Text4
// Text
// Text2
// Text3
Text_t Example;
Example += uT("Text0");
Example += uT("Unknown");
Example += uT("Text1");
Example += uT("Unknown");
Example += uT("Unknown");
Example += uT("Text2");
Example += uT("Unknown");
Example += uT("Unknown");
Example += uT("Unknown");
Example += uT("Text3");
// Удаление первой строки
Example -= 0;
// Удаление отдельных строк 'Unknown'
Example = Utilities::Remove(Example,
[](const auto & _Line)
{
return (_Line.Value.To<String_t>() == uT("Unknown"));
});
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
// Удаление блока текста
Text_t Example;
Example += uT("Text1");
// Будет удален этот блок
Example += uT("Begin");
Example += uT("Text2");
Example += uT("Text3");
Example += uT("End");
// //////////////////////
Example += uT("Text2");
Example += uT("Text3");
bool IsInsideBlock = false;
Example = Utilities::Remove(Example,
[&](const auto & _Line)
{
if (_Line.Value.To<String_t>() == uT("Begin")) IsInsideBlock = true;
bool IsRemoved = IsInsideBlock;
if (_Line.Value.To<String_t>() == uT("End")) IsInsideBlock = false;
return IsRemoved;
});
// В результате будет текстовый документ, содержащий
// Text1
// Text2
// Text3
