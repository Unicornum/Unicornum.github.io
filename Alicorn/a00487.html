Подробное описание

Классы для работы с документами xml.

Пример использования:

// Предварительные объявления
using Xml_t = ::alicorn::document::Xml<XML_IMPLEMENTATION>;
using Path_t = boost::filesystem::path;
using Data_t = std::vector<boost::int8_t>;
using String_t = alicorn::extension::std::String;
// Создание документа из строки
const std::string RawData =
"\xEF\xBB\xBF"
"<?xml version=\"1.0\" encoding=\"utf-8\"?>\r\n"
"<entity1>\r\n"
" <entity2/>\r\n"
" <entity3/>\r\n"
" <entity4/>\r\n"
"</entity1>\r\n";
const Data_t Data(RawData.begin(), RawData.end());
Xml_t Example(Data);
const Xml_t & ConstExample = Example;
using namespace alicorn::document::xml;
// Переход к узлу entity1
Example[Root].CurrentNode[_T("entity1")];
Count = 0;
// Перебор всех дочерних узлов узла entity1
auto & Node = Example.CurrentNode;
for (bool IsExists = Node[FirstChild]; IsExists; IsExists = Node[Next])
{
Count++;
}
EXPECT_EQ(3, Count);
// Добавление дочернего узла текущему
Example.CurrentNode += _T("entity");
// Получение значения узла
String_t Empty = Example[Root].CurrentNode[_T("params")].Value.To<String_t>();
// Изменение значения узла
Example[Root].CurrentNode[_T("params")][_T("param")].Value =
_T("another_value");
// Получение значения атрибута с именем param1 у текущего узла
String_t Value1 = Example.CurrentNode.Attributes[_T("param1")].To<String_t>();
// Изменение значения атрибута с именем param1 у текущего узла
Example.CurrentNode.Attributes[_T("param1")] = _T("value0");
