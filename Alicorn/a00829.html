Подробное описание

Раздел классов реализации виртуальной файловой системы.

Использование виртуальной файловой системы

Получение данных

#include <alicorn\vfs.hpp>
#include <alicorn\converter.hpp>
using ConverterPtr_t = ::std::shared_ptr<::alicorn::vfs::IConverter>;
using Converters_t = ::std::vector<ConverterPtr_t>;
// Путь к текущей программе
using CurrentModule_t = ::alicorn::system::CurrentModule;
const auto ModuleLocation = CurrentModule_t::GetLocation();
// Объявления используемых классов
using Cryptor_t = ::alicorn::converter::Crypto<::alicorn::converter::AES256>>;
using Password_t = ::Password; // Подробнее см. далее.
using Compressor_t = ::alicorn::converter::Compress<::alicorn::converter::Snappy>>;
// Объект виртуальной файловой системы
::alicorn::vfs::Core Vfs(
{
// Объект для поиска данных в файле в папке программы.
::std::make_shared<::alicorn::vfs::FileSystem>(ModuleLocation),
// Объект для поиска данных в файле в папке данных программы.
::std::make_shared<::alicorn::vfs::FileSystem>(CurrentModule_t::GetAppDataPath()),
// Объект для поиска данных внутри упакованного файла Main.pack в папке
// программы (он должен содержать данные, упакованные тем же способом).
// Пример утилиты упаковщика для этого примера см. ниже.
::std::make_shared<::alicorn::vfs::Pack>(ModuleLocation / L"Main.pack",
Converters_t
{
// Использовать декодировщик данных, использующий пароль из protect.dll в
// папке программы.
::std::make_shared<::alicorn::vfs::Crypto<Cryptor_t, Password_t>>(ModuleLocation / L"protect.dll"),
// Использовать указанный способ сжатия данных.
::std::make_shared<::alicorn::vfs::Compress<Compressor_t>>(),
}),
});
// Получение данных:
// SkyBoxData будет содержать данные из того файла SkyBox.png, который будет
// найден первым при поиске в следующем порядке:
// - папка Texture в папке программы,
// - папка Texture в папке данных программы,
// - папка Texture внутри упакованного файла Main.pack.
const auto SkyBoxData = Vfs.GetData(L"Texture\\SkyBox.png");

Использование dll для получения пароля

Для шифрования/дешифрования данных внутри .pack требуется пароль, который должен формироваться внутри специально созданной protect.dll, например, так:

#include "stdafx.h"
#include <vector>
#include <boost\filesystem.hpp>
#include <alicorn\vfs.hpp>
#include "Seed.hpp"
#define EXPORT __declspec(dllexport)
// Вспомогательный макрос для объявления пароля.
#define Ec(Char) Encryptor<Char>::Int
class Password final :
public ::alicorn::vfs::IPassword
{
template <char Char>
class Encryptor
{
public:
enum
{
// Способ кодирования пароля.
Int = (Char << (PASSWORD_SEED & 0x1F)) ^ PASSWORD_SEED
};
};
public:
EncryptPassword_t GetPassword(void) const override
{
return
{
// Пароль в явном виде (он кодируется на стадии компиляции, поэтому
// внутри dll не будет ни самого пароля, ни способа его кодирования).
Ec('p'),
Ec('a'),
Ec('s'),
Ec('s'),
Ec('w'),
Ec('o'),
Ec('r'),
Ec('d'),
};
}
};
extern "C"
{
EXPORT ::alicorn::system::dll::Object * CreateObjectFromDll(void)
{
return new Password;
}
EXPORT void Destroy(::alicorn::system::dll::Object * _pObject)
{
delete _pObject;
}
}

Файл Seed.hpp

#pragma once
// Ключевое число, используемое для шифрования пароля.
static const int32_t PASSWORD_SEED = 0x12345678;

Класс для дешифровки пароля

class Password final
{
using EncryptPassword_t = std::vector<int32_t>;
static const int32_t SEED = PASSWORD_SEED;
public:
static ::std::string Decrypt(const EncryptPassword_t & _Password)
{
::std::string Result;
for (const auto & Element : _Password)
{
// Действие, обратное тому, что используется внутри protect.dll
Result += static_cast<char>((Element ^ SEED) >> (SEED & 0x1F));
}
return Result;
}
};

Пример реализации утилиты упаковки содержимого папки в .pack

#include <alicorn\vfs.hpp>
#include <alicorn\converter.hpp>
int _tmain(int argc, _TCHAR * argv[])
{
if (argc != 4) return 1;
// argv[1] - путь к папке, которая будет упаковываться.
// argv[2] - путь к protect.dll, содержащей пароль.
// argv[3] - путь к файлу, в который будет записан результат упаковки.
// Объявления используемых классов
using Cryptor_t = ::alicorn::converter::Crypto<::alicorn::converter::AES256>;
using Password_t = ::Password; // См. выше.
using Compressor_t = ::alicorn::converter::Compress<::alicorn::converter::Snappy>;
// Объект для упаковки данных в файл Main.pack в папке программы
::alicorn::vfs::Pack Packer(argv[3],
{
// Использовать кодировщик данных, использующий пароль из protect.dll в
// папке программы.
::std::make_shared<::alicorn::vfs::Crypto<Cryptor_t, Password_t>>(argv[2]),
// Использовать указанный способ сжатия данных.
::std::make_shared<::alicorn::vfs::Compress<Compressor_t>>(),
});
// Упаковка данных из указанной папки в файл, указанный в конструкторе
Packer.Build(argv[1]);
return 0;
}
