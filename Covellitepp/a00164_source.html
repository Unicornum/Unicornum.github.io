101 template<class T>
102 auto GraphicApi::Constants::GetBuffer(void) -> Data<T> &
103 {
104  static const auto hCode = typeid(T).hash_code();
105 
106  const auto itConstant = m_Data.find(hCode);
107  if (itConstant == m_Data.end())
108  {
109  throw STD_EXCEPTION << "Unexpected type: " << typeid(T).name();
110  }
111 
112  return *::covellite::any_cast<DataPtr_t<T> &>(itConstant->second);
113 }
