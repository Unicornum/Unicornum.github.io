28Initializer::Initializer(const Data & _Data) :
29 m_System(_Data.m_pStringTranslator),
30 m_pRenderInterface(_Data.m_pRenderInterface)
31{
32 CovelliteGui::Core::SetFileInterface(&m_File);
33 CovelliteGui::Core::SetSystemInterface(&m_System);
34 CovelliteGui::Core::SetRenderInterface(m_pRenderInterface.get());
35
36 // Инициализация сделана через статический объект из-за того, что
37 // CovelliteGui::Core::Initialise() в рамках одного модуля можно запускать
38 // только один раз (CovelliteGui::Core::Shutdown() не поможет, повторная
39 // инициализация даже в этом случае приводит к падению программы),
40 // а android-приложение загружает главный .so модуль только один раз при
41 // первом старте программы, а затем при каждой активации программы просто
42 // вызывает функцию этого модуля.
43 Helper::GetInstance();
44}
